A collection of common array problems with code examples.

1. Find Smallest and Largest Number

Finds the minimum and maximum values in a given array of integers.


// Java Solution
import java.util.Arrays;
public class ArrayProblems {
    public static void findMinMax(int[] arr) {
        if (arr == null || arr.length == 0) {
            System.out.println("Array is empty or null.");
            return;
        }
        int min = arr[0];
        int max = arr[0];
        for (int i = 1; i < arr.length; i++) {
            if (arr[i] < min) { min = arr[i]; }
            if (arr[i] > max) { max = arr[i]; }
        }
        System.out.println("Smallest: " + min);
        System.out.println("Largest: " + max);
    }
    public static void main(String[] args) {
        // Input:
        int[] numbers = {5, 8, 2, 10, 3};
        findMinMax(numbers);
        // Output:
        // Smallest: 2
        // Largest: 10
    }
}

2. Find Missing Number in a Shuffled Array

Finds the single missing number in an array of consecutive integers from 1 to N.


// Java Solution
public class ArrayProblems {
    public static int findMissingNumber(int[] arr) {
        int n = arr.length + 1;
        int expectedSum = n * (n + 1) / 2;
        int actualSum = 0;
        for (int num : arr) { actualSum += num; }
        return expectedSum - actualSum;
    }
    public static void main(String[] args) {
        // Input:
        int[] numbers = {1, 2, 4, 5, 6};
        // Output:
        System.out.println("Missing number: " + findMissingNumber(numbers));
    }
}
                    

# Python Solution
def find_missing_number(arr):
    n = len(arr) + 1
    expected_sum = n * (n + 1) // 2
    actual_sum = sum(arr)
    return expected_sum - actual_sum

# Example
# Input:
numbers = [1, 2, 4, 5, 6]
# Output:
print(f"Missing number: {find_missing_number(numbers)}")

3. Find Second Largest Number

Identifies the second largest value in an array in a single pass.


// Java Solution
public class ArrayProblems {
    public static int findSecondLargest(int[] arr) {
        if (arr == null || arr.length < 2) { return -1; }
        int largest = Integer.MIN_VALUE;
        int secondLargest = Integer.MIN_VALUE;
        for (int num : arr) {
            if (num > largest) {
                secondLargest = largest;
                largest = num;
            } else if (num > secondLargest && num != largest) {
                secondLargest = num;
            }
        }
        return secondLargest;
    }
    public static void main(String[] args) {
        // Input:
        int[] numbers = {12, 35, 1, 10, 34, 1};
        // Output:
        System.out.println("Second largest: " + findSecondLargest(numbers));
    }
}

4. Re-arranged Array by Small-Large Combination

Sorts an array and then prints it by alternating the smallest and largest elements.


// Java Solution
import java.util.Arrays;
public class ArrayProblems {
    public static void printSmallLargeCombination(int[] arr) {
        Arrays.sort(arr);
        int i = 0;
        int j = arr.length - 1;
        while (i <= j) {
            if (i == j) { System.out.print(arr[i] + " "); }
            else { System.out.print(arr[i] + " " + arr[j] + " "); }
            i++;
            j--;
        }
    }
    public static void main(String[] args) {
        // Input:
        int[] numbers = {1, 2, 3, 4, 5, 6};
        // Output:
        printSmallLargeCombination(numbers);
    }
}
                    

# Python Solution
def print_small_large_combination(arr):
    arr.sort()
    i, j = 0, len(arr) - 1
    while i <= j:
        if i == j: print(arr[i], end=" ")
        else: print(arr[i], arr[j], end=" ")
        i += 1
        j -= 1

# Example
# Input:
numbers = [1, 2, 3, 4, 5, 6]
# Output:
print_small_large_combination(numbers)

5. Max Contiguous Sequence

Finds the maximum sum of a contiguous sub-array (Kadane's algorithm).


// Java Solution
public class ArrayProblems {
    public static int maxContiguousSequenceSum(int[] arr) {
        int maxSoFar = Integer.MIN_VALUE;
        int maxEndingHere = 0;
        for (int num : arr) {
            maxEndingHere += num;
            if (maxSoFar < maxEndingHere) { maxSoFar = maxEndingHere; }
            if (maxEndingHere < 0) { maxEndingHere = 0; }
        }
        return maxSoFar;
    }
    public static void main(String[] args) {
        // Input:
        int[] numbers = {-2, -3, 4, -1, -2, 1, 5, -3};
        // Output:
        System.out.println("Max sum: " + maxContiguousSequenceSum(numbers));
    }
}

6. Print the Most Repeated Number

Counts frequencies to find the number that appears most often.


// Java Solution
import java.util.HashMap;
import java.util.Map;
public class ArrayProblems {
    public static int findMostRepeated(int[] arr) {
        Map frequencyMap = new HashMap<>();
        for (int num : arr) {
            frequencyMap.put(num, frequencyMap.getOrDefault(num, 0) + 1);
        }
        int mostRepeated = -1;
        int maxCount = 0;
        for (Map.Entry entry : frequencyMap.entrySet()) {
            if (entry.getValue() > maxCount) {
                maxCount = entry.getValue();
                mostRepeated = entry.getKey();
            }
        }
        return mostRepeated;
    }
    public static void main(String[] args) {
        // Input:
        int[] numbers = {1, 2, 2, 3, 3, 3, 4, 5};
        // Output:
        System.out.println("Most repeated: " + findMostRepeated(numbers));
    }
}
                    

# Python Solution
from collections import Counter
def find_most_repeated(arr):
    if not arr: return None
    counts = Counter(arr)
    return max(counts, key=counts.get)

# Example
# Input:
numbers = [1, 2, 2, 3, 3, 3, 4, 5]
# Output:
print(f"Most repeated: {find_most_repeated(numbers)}")
                    

// C# Solution
using System;
using System.Linq;
public class ArrayProblems {
    public static int FindMostRepeated(int[] arr) {
        if (arr == null || arr.Length == 0) { return -1; }
        return arr.GroupBy(x => x).OrderByDescending(g => g.Count()).First().Key;
    }
    public static void Main(string[] args) {
        // Input:
        int[] numbers = {1, 2, 2, 3, 3, 3, 4, 5};
        // Output:
        Console.WriteLine($"Most repeated: {FindMostRepeated(numbers)}");
    }
}

7. Print All Unique Elements

Uses a set to get a collection of unique numbers.


// Java Solution
import java.util.HashSet;
import java.util.Set;
public class ArrayProblems {
    public static void printUniqueElements(int[] arr) {
        Set uniqueElements = new HashSet<>();

        // Input:

        // Output:

8. Rotate an Array by 'N' Positions

Rotates the array elements by a specified number of positions.


// Java Solution
import java.util.Arrays;
public class ArrayProblems {
    public static void rotateArray(int[] arr, int n) {
        n = n % arr.length;
        if (n < 0) { n += arr.length; }
        int[] temp = new int[arr.length];
        for (int i = 0; i < arr.length; i++) {
            temp[(i + n) % arr.length] = arr[i];
        }
        System.arraycopy(temp, 0, arr, 0, arr.length);
    }
    public static void main(String[] args) {
        // Input:
        int[] numbers = {1, 2, 3, 4, 5, 6, 7};
        int n = 3;
        rotateArray(numbers, n);
        // Output:
        System.out.println("Rotated array: " + Arrays.toString(numbers));
    }
}
                    

# Python Solution
def rotate_array(arr, n):
    n = n % len(arr)
    return arr[-n:] + arr[:-n]

# Example
# Input:
numbers = [1, 2, 3, 4, 5, 6, 7]
n = 3
rotated = rotate_array(numbers, n)
# Output:
print(f"Rotated array: {rotated}")
                    

// C# Solution
using System;
using System.Linq;
public class ArrayProblems {
    public static void RotateArray(int[] arr, int n) {
        n = n % arr.Length;
        if (n < 0) { n += arr.Length; }
        Array.Reverse(arr);
        Array.Reverse(arr, 0, n);
        Array.Reverse(arr, n, arr.Length - n);
    }
    public static void Main(string[] args) {
        // Input:
        int[] numbers = {1, 2, 3, 4, 5, 6, 7};
        int n = 3;
        RotateArray(numbers, n);
        // Output:
        Console.WriteLine("Rotated array: " + string.Join(", ", numbers));
    }
}

9. Remove All Duplicates

Creates a new array containing only the unique elements from the original.


// Java Solution
import java.util.Arrays;
import java.util.HashSet;
import java.util.Set;
public class ArrayProblems {
    public static int[] removeDuplicates(int[] arr) {
        Set uniqueElements = new HashSet<>();
        for (int num : arr) { uniqueElements.add(num); }
        int[] result = new int[uniqueElements.size()];
        int i = 0;
        for (int num : uniqueElements) { result[i++] = num; }
        return result;
    }
    public static void main(String[] args) {
        // Input:
        int[] numbers = {1, 2, 2, 3, 4, 4, 5};
        int[] uniqueArray = removeDuplicates(numbers);
        // Output:
        System.out.println("Array without duplicates: " + Arrays.toString(uniqueArray));
    }
}

10. Move all 0s to the end of an array

Moves all zeros to the end of the array while maintaining the relative order of non-zero elements.


// Java Solution
import java.util.Arrays;
public class ArrayProblems {
    public static void moveZeros(int[] arr) {
        int nonZeroIndex = 0;
        for (int num : arr) {
            if (num != 0) {
                arr[nonZeroIndex++] = num;
            }
        }
        while (nonZeroIndex < arr.length) {
            arr[nonZeroIndex++] = 0;
        }
    }
    public static void main(String[] args) {
        // Input:
        int[] numbers = {1, 0, 2, 0, 3, 0, 4};
        moveZeros(numbers);
        // Output:
        System.out.println("Array with zeros moved: " + Arrays.toString(numbers));
    }
}
                    

# Python Solution
def move_zeros(arr):

print(f"Array with zeros moved: {numbers}")
